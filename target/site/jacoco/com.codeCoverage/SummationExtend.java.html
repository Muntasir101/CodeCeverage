package com.codeCoverage;

public class SummationExtend {

    public int doSum(int a, int b){
        int sum=a+b;
        if(sum>100){
            System.out.println("Its DONE");
        }
        else if(a>50){
            System.out.println("Its PENDING");
        }
        return sum;
    }
}
